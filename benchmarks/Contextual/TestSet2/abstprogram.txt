int main() {
    Set sPos = new(); // positive set
    Set sNeg = new(); // negative set
    int N = *;
    // min1 = MAX, max2 = MIN;

    for (int i = 0; i < N; i++) { // inv1 (N, min1, max2)
        int v1 = nondet();
        if (v1 >= 0) {
            sPos.insert(v1); // insert1(v1, min1)
        }else{
        sNeg.insert(v1); // insert2(v2, max2)
	}
    }

    int sum1 = sPos.computeSum(); // computeSum1(min1, max2, sum1)
    int sum2 = sNeg.computeSum(); // computeSum2(min1, max2, sum2)
    assert(sum2 < 0 and sum1 >= 0); 
}
