Distances dist; 
NodeMap nodes; 
PriorityType previous_distance = -INFINITY; 

FibonacciHeap pq;

initialize_all_nodes(graph, dist, nodes, pq, start_node);
dist[start_node] = 0;
pq.decrease_key(nodes[start_node], 0);

while (!pq.is_empty()) {

    key, distance = pq.extract_min();
    assert(distance >= previous_distance);

    previous_distance = distance;

    while (*) { 
        v, w = get_neighbor(key); 
        
        alt_distance = distance + w;
        if (alt_distance < dist[v]) { 
            assert(alt_distance < dist[v]); 
            dist[v] = alt_distance;
            pq.decrease_key(nodes[v], alt_distance);
        }
    }
}
